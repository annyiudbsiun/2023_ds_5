// 必要なライブラリを読み込む
const canvas = document.querySelector("canvas");
const ctx = canvas.getContext("2d");
// ゲームの状態を管理するオブジェクトを作成
class Game {
constructor() {
this.score = 0;
this.lives = 3;
this.enemies = [];
this.bullets = [];
this.spawnEnemy();
}
// 敵を生成
spawnEnemy() {
this.enemies.push({
x: 100,
y: 0,
type: Math.floor(Math.random() * 3),
});
}
// 敵を移動
moveEnemies() {
for (let i = 0; i < this.enemies.length; i++) {
const enemy = this.enemies[i];
enemy.y += 10;
if (enemy.y > canvas.height) {
this.lives--;
if (this.lives <= 0) {
this.gameOver();
}
this.spawnEnemy();
}
}
}
// 弾を移動
moveBullets() {
for (let i = 0; i < this.bullets.length; i++) {
const bullet = this.bullets[i];
bullet.y -= 10;
if (bullet.y < 0) {
this.bullets.splice(i, 1);
}
}
}
// 弾と敵の衝突判定
collision() {
for (let i = 0; i < this.bullets.length; i++) {
const bullet = this.bullets[i];
for (let j = 0; j < this.enemies.length; j++) {
const enemy = this.enemies[j];
if (bullet.x >= enemy.x && bullet.x <= enemy.x + 32 && bullet.y
>= enemy.y && bullet.y <= enemy.y + 32) {
this.score += 10;
this.enemies.splice(j, 1);
this.bullets.splice(i, 1);
if (enemy.type === 0) {
this.spawnEnemy();
}
}
}
}
}
// ゲームオーバー処理
gameOver() {
console.log("ゲームオーバー");
}
